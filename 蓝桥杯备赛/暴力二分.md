## [P9240 [蓝桥杯 2023 省 B] 冶炼金属 - 洛谷](https://www.luogu.com.cn/problem/P9240)
自己想的区间约束求解
```cpp
const ll MAXN = 1e18+1;
int n;
struct v{
    ll l,r;
};

void solve() {
    int n;cin>>n;
    vector<v> a(n);
    for(int i=0;i<n;i++){
        ll o,x,l,r;cin>>o>>x;
        r=o/x;
        l=(o+x)/(x+1);
        a[i].l=l;
        a[i].r=r;
    }
    ll left=0,right=MAXN;
    for(int i=0;i<n;i++){
        left=max(left,a[i].l);
        right=min(right,a[i].r);
    }
    cout<<left<<" "<<right;
}
```
## [P9244 [蓝桥杯 2023 省 B] 子串简写 - 洛谷](https://www.luogu.com.cn/problem/P9244)
自己第一次做犯了两个错误，第一个一开始逻辑反了过来固定c2找c1，lower_bound只能找左范围，因此会漏解，第二个是忽视了c1和c2相同的错误
```cpp
int n;
string s;
char a,b;
vector<int> aa,bb;

void solve() {
	cin>>n>>s>>a>>b;
    for(int i=0;i<s.size();i++){
        if(s[i]==a){
            aa.push_back(i);
        }
        if(s[i]==b){
            bb.push_back(i);
        }
    }
    ll ans=0;
    for(int i=0;i<aa.size();i++){
        auto l=lower_bound(bb.begin(),bb.end(),aa[i]+n-1);
        ans+=bb.size()-(l-bb.begin());
    }
    cout<<ans;
}
```
