# [U549625 小苯的区间和疑惑 - 洛谷](https://www.luogu.com.cn/problem/U549625)
```cpp
int n;

void solve() {
	cin>>n;
	vector<int> a(n+1,0);
	vector<ll> pre(n+1,0);
	for(int i=1;i<=n;i++){
		cin>>a[i];
		pre[i]=a[i]+pre[i-1];
	}
	for(int i=1;i<=n;i++){
		ll ans=0;
		if(i<n)
		for(int k=i+1;k<=n;k++){
			ans=max(ans,pre[k]-pre[i-1]);
			for(int j=1;j<i;j++){
				ans=max({ans,pre[k]-pre[j-1],pre[i]-pre[j-1]});
			}
		}
		else{
			for(int j=1;j<i;j++){
				ans=max({ans,pre[i]-pre[j-1]});
			}
		}
		cout<<ans<<' ';
	}
}
```
