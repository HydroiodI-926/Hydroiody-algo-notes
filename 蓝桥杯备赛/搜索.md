## [P9243 [蓝桥杯 2023 省 B] 岛屿个数 - 洛谷](https://www.luogu.com.cn/problem/P9243)
dfs搜外岛 bfs搜外水
```cpp
#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

int n,m,ans;
vector<vector<int>> mp;
vector<vector<bool>> vis;
struct p{
	int x,y;
};
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
int hx[]={0,1,1,1,0,-1,-1,-1};
int hy[]={1,1,0,-1,-1,-1,0,1};

void dfs(int x,int y){
	vis[x][y]=1;
	for(int i=0;i<4;i++){
		int nx=x+dx[i],ny=y+dy[i];
		if(nx>0&&nx<=n&&ny>0&&ny<=m
		&&mp[nx][ny]==1&&!vis[nx][ny]){
			dfs(nx,ny);
		}
	}
}

void bfs(int x,int y){
	queue<p> q;
	vis[x][y]=1;
	q.push({x,y});
	while(!q.empty()){
		auto [a,b]=q.front();
		q.pop();
		for(int i=0;i<8;i++){
			int nx=a+hx[i],ny=b+hy[i];
			if(nx>0&&nx<=n&&ny>0&&ny<=m
				&&mp[nx][ny]==1&&!vis[nx][ny]){
				ans++;
				dfs(nx,ny);
			}
			else if(nx>0&&nx<=n&&ny>0&&ny<=m
				&&mp[nx][ny]==0&&!vis[nx][ny]){
				vis[nx][ny]=1;
				q.push({nx,ny});
			}
		}
	}
}

void solve() {
	cin>>n>>m;
	ans=0;
	mp.assign(n+1,vector<int>(m+1,0));
	vis.assign(n+1,vector<bool>(m+1,0));
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			char c;cin>>c;
			mp[i][j]=c-'0';
		}
	}
	for(int i=1;i<=m;i++){
		if(mp[1][i]==0&&!vis[1][i]){
			bfs(1,i);
		}
		if(mp[1][i]==1&&!vis[1][i]){
			ans++;
			dfs(1,i);
		}
	}
	for(int i=1;i<=n;i++){
		if(mp[i][1]==0&&!vis[i][1]){
			bfs(i,1);
		}
		if(mp[i][1]==1&&!vis[i][1]){
			ans++;
			dfs(i,1);
		}
	}
	for(int i=1;i<=n;i++){
		if(mp[i][m]==0&&!vis[i][m]){
			bfs(i,m);
		}
		if(mp[i][m]==1&&!vis[i][m]){
			ans++;
			dfs(i,m);
		}
	}
	for(int i=1;i<=m;i++){
		if(mp[n][i]==0&&!vis[n][i]){
			bfs(n,i);
		}
		if(mp[n][i]==1&&!vis[n][i]){
			ans++;
			dfs(i,1);
		}
	}
	cout<<ans<<'\n';
}
```
