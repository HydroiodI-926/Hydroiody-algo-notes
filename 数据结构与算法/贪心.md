# 固定一个区间找最值
针对于一些数据量在$10^9$次方的题不能模拟或者遍历去过题
数据量较大，不能通过dp去记录所有可能的值，只能通过贪心，但是在分析题目的时候，分类讨论，分析不同策略的优劣进而找到一种绝对有利的方式没有什么意义，完全可以把大数据分成大部分和小部分，大部分可以遵从直觉，选择出宏观上最优解，小部分的可选择的策略很少，完全可以==枚举策略==，这时只需要注意遍历所有可能的边界情况和所有的贪心策略（一般只有几种），就像是在二次函数的极值附近找极大值或极小值
## [U665271 F 新年新偷菜 - 洛谷](https://www.luogu.com.cn/problem/U665271?contestId=313069)

- 由于x和y的差距可能很小，分类讨论需要花时间造特定数据，而往往答案要么取大部分ALL IN x或ALL IN y，然后剩下小部分只有有限个x方式和y方式的组合策略，暴力枚举即可
- 两种策略其实是偷2种和偷3种的方式，找到它们的最小公倍数6作为小部分，再偷6个菜的时候两种方式的结果是一样的，这时候容易比较两种方式的微小溢出和浪费
```cpp
const ll MAXN = 2e18;
ll n,x,y,ans=MAXN;

void solve() {
	cin>>n>>x>>y;
	int k=n/3;
	for(int i=0;i<=min(2,k);i++){
		ll s=n-i*3;
		ans=min(ans,i*y+(s+1)/2*x);
	}
	for(int i=max(0,k-2);i<=k;i++){
		ll s=n-i*3;
		ans=min(ans,i*y+(s+1)/2*x);
	}
	cout<<ans;
}
```

# 利用优先队列解决问题
## [1.最大开支 - 蓝桥云课](https://www.lanqiao.cn/problems/3541/learning/?page=1&first_category_id=1&name=%E6%9C%80%E5%A4%A7%E5%BC%80%E6%94%AF)
```cpp
struct eq{
	ll k,b,cnt;
	
	ll total(ll x) const{
		ll p=max(k*x+b,0LL);
		return p*x;
	}
	
	ll next() const{
		return total(cnt+1)-total(cnt);
	}
	
	bool operator<(const eq& other) const{
		return next()<other.next();
	}
};

void solve(){
	int n,m;
	cin>>n>>m;
	priority_queue<eq> pq;
	for(int i=0;i<m;i++){
		ll k,b;cin>>k>>b;
		pq.push({k,b,0});
	}
	ll ans=0;
	for(int i=0;i<n;i++){
		eq t=pq.top();
		pq.pop();
		ll gain=t.next();
		if(gain<=0) break;
		ans+=gain;
		t.cnt++;
		pq.push(t);
	}
	cout<<ans;
}
```

```cpp
int n,k;
void solve(){
	cin>>n>>k;
	vector<int> a(n);
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a.begin(),a.end(),greater<int>());
	int b=0,m=0,re=0,ans=0;
	for(int i=0;i<n;i++){
		m++;
		int ew=(a[i]-1+re)/k;
		if(m+b+ew>=n){
			ans+=max(0,(n-m-b)*k-re)+1;
			break;
		} 
		else{
			ans+=a[i];
			b+=ew;
			re=(a[i]-1+re)%k;
		}
	}
	cout<<ans<<"\n";
}
```

# 反悔贪心
```cpp
int n,m,k;
string s;


void solve(){
    cin>>n>>m>>k>>s;
    ll tb=m,p=0,cnta=0,ans=0;
    for(char c:s){
    	if(c=='I'){
    		if(tb>0){
    			tb--;
    			p+=(k-1);
    			ans++;
    		}
    	}
    	else if(c=='A'){
    		if(p>0){
    			p--;
    			cnta++;
    			ans++;
    		}
    		else if(tb>0){
    			tb--;
    			p+=(k-1);
    			ans++;
    		}
    	}
    	else if(c=='E'){
    		if(p>0){
    			p--;
    			ans++;
    		}
    		else if(cnta>0&&tb>0){
    			tb--;
    			p+=(k-1);
    			cnta--;
    			ans++;
    		}
    	}
    }
    cout<<ans<<"\n";
}
```
