# 图论
## 反向边建图
[P3916 图的遍历 - 洛谷](https://www.luogu.com.cn/problem/P3916#ide)
```cpp
const int N = 1e5+5;

int n,m;
vector<int> adj[N];
int a[N];

void dfs(int s,int ans){
	a[s]=ans;
	for(int v:adj[s]){
		if(a[v]==0){
			dfs(v,ans);
		}
	}
}

void solve() {
    cin>>n>>m;
    for(int i=0;i<m;i++){
    	int a,b;
    	cin>>a>>b;
    	adj[b].push_back(a);
    }
    for(int i=n;i>=0;i--){
    	if(a[i]==0){
    		dfs(i,i);
    	}
    }
    for(int i=1;i<=n;i++){
    	cout<<a[i]<<' ';
    }
}
```
