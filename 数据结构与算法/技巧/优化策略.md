[Open Credit System - UVA 11078 - Virtual Judge](https://vjudge.net/problem/UVA-11078)
```cpp
int n;
void solve(){
	cin>>n;
	vector<ll> a(n);
	for(int i=0;i<n;i++){
		cin>>a[i];
	} 
	ll ans=-MAXN;
	ll ma=a[0];
	for(int i=1;i<n;i++){
		ans=max(ans,ma-a[i]);
		ma=max(ma,a[i]);
	}
	cout<<ans<<"\n";
}
```

[Subsequence - POJ 3061 - Virtual Judge](https://vjudge.net/problem/POJ-3061)
有明显单调性，前缀和/滑动窗口
```cpp
void solve(){
	cin>>n>>k;
	vector<ll> a(n+1,0);
	for(int i=1;i<=n;i++){
		cin>>a[i];
		a[i]+=a[i-1];
	}
	ll ans=MAXN;
	for(int i=0,j=1;j<=n;j++){
		if(a[j]-a[i]>=k){
			while(a[j]-a[i]>=k) i++;
			ans=min(ans,ll(j-i+1));	
		}
	}
	cout<<(ans==MAXN?0:ans)<<endl;
}

void solve(){
	cin>>n>>k;
	vector<ll> a(n);
	for(int i=0;i<n;i++){
		cin>>a[i];
	} 
	int r=-1;
	ll sum=0,ans=MAXN;
	for(ll l=0;l<n;l++){
		while((r<l||sum<k)&&r<n-1){
			sum+=a[++r];
		}
		if(sum>=k)
			ans=min(ans,r-l+1);
		sum-=a[l];
	}
	cout<<(ans==MAXN?0:ans)<<endl;
}
```
