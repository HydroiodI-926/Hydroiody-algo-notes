前置：[[排序 归并排序]]

- 利用归并排序的过程顺便得到答案，以便减少减少复杂度（o(n^2)->o(nlogn)）的框架
- 也是一种算贡献的算法框架
## 原理
1. 一个问题的答案如果在一个大范围上，且等于 左部分答案+右部分答案+跨越左右产生的答案
2. 计算跨越左右产生的答案时，如果加上左右有序，是否得到左右的便利性
3. 满足以上两点，在归并排序的框架中加入求解答案的逻辑很可能解决
## 例题
### [计算数组的小和](https://www.nowcoder.com/practice/edfe05a1d45c4ea89101d936cac32469)
#### 解法
-  通过排序产生了计算便利性，排序完了，答案也顺便求的了
```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll MAXN = 1e5;

int n;
vector<int> arr;
vector<int> help;

ll merge(int l,int m,int r){
	ll ans=0;
	for(int j=m+1,i=l,sum=0;j<=r;j++){
		while(i<=m&&arr[i]<=arr[j]){
			sum+=arr[i++];
		}
		ans+=sum;
	}
	int i=l,a=l,b=m+1;
	while(a<=m&&b<=r){
		help[i++]=arr[a]<=arr[b]?arr[a++]:arr[b++];
	}
	while(a<=m){
		help[i++]=arr[a++];
	}
	while(b<=r){
		help[i++]=arr[b++];
	}
	for(i=l;i<=r;i++){
		arr[i]=help[i];
	}
	return ans;
}

ll smallSum(int l,int r){
	if(l==r){
		return 0;
	}
	int m=(l+r)/2;
	return smallSum(l,m)+smallSum(m+1,r)+merge(l,m,r);
}

void solve(){
	cin>>n;
	arr.resize(n);
	help.assign(n,0);
	for(int &v:arr){
		cin>>v;
	}
	cout<<smallSum(0,n-1);
}

int main(){
	ios::sync_with_stdio(0);
	cin.tie(0),cout.tie(0);
	solve();
	// int T;cin>>T;
    // while(T--){
    	// solve();
    // }
	return 0;
}
```

### [P1908 逆序对 - 洛谷](https://www.luogu.com.cn/problem/P1908)
把原数组拆成ab两个小数组b数组原本在a数组右边，如果b数组的数比a数组小，那么b数组的这个数就会算成一个逆序对贡献，每次排a数组的数的时候加上此时的贡献c，也就是a数组的这个数能贡献几个逆序对
```cpp
int n;
ll cnt;
vector<ll> arr,help;

void merge(int l,int m,int r){
	int i=l,a=l,b=m+1;
	ll c=0;  // 计算逆序对数
	while(a<=m&&b<=r){
		if(arr[a]>arr[b]){
			help[i]=arr[b];
			b++;
			c++; // b数组的数如果比a小，那么就会添加一个贡献
		}
		else{
			help[i]=arr[a];
			a++;
			cnt+=c; 
		}
		i++;
	}
	while(a<=m){
		help[i++]=arr[a++];
		cnt+=c;
	}
	while(b<=r){
		help[i++]=arr[b++];
	}
	for(int i=l;i<=r;i++){
		arr[i]=help[i];
	}
	
}

void ms(int l,int r){
	if(l==r){
		return;
	}
	int m=l+(r-l)/2;
	ms(l,m);
	ms(m+1,r);
	merge(l,m,r);
}

void solve() {
	cin>>n;
	arr.resize(n);
	help.resize(n);
	for(int i=0;i<n;i++){
		cin>>arr[i];
	}
	ms(0,n-1);
	cout<<cnt;
}
```

## Problem Statement

You are given a sequence of integers  $$ A = (A_1, A_2, \dots, A_N) $$of length \( N \).
Define \( f(l, r) \) as:
- the number of distinct values in the subsequence $$ (A_l, A_{l+1}, \dots, A_r) $$Evaluate the following expression:$$
\sum_{i=1}^{N}\sum_{j=i}^{N} f(i, j).
$$
## Constraints

$$ 1 \le N \le 2 \times 10^{5} $$
$$ 1 \le A_i \le N $$
- All input values are integers.
## Input
The input is given from Standard Input in the following format:
- \( N \)
- $$( A_1\ \dots\ A_N )$$
## Output

Print the answer.
## Sample Input 1

13 14
