# 二叉树遍历
## 递归遍历dfs
- 先序遍历 中左右
- 中序遍历 左中右
- 后序遍历 左右中
二叉树上dfs的实现，基本方向是从上到下，从左到右，递归遍历顾名思义就是通过递归实现的遍历，代码基本架构都一样，只是打印的时机不同
```cpp
// 二叉树定义
struct TreeNode{
	int val;
	TreeNode *left;
	TreeNode *right;
	TreeNode():val(0),left(nullptr),right(nullptr){}
	TreeNode(int x): val(x),left(nullptr),right(nullptr){}
	TreeNode(int x,TreeNode *left,TreeNode *right):val(x),left(left),right(right){}
};

// 递归版
// 递归遍历代码
void dfs(TreeNode* p){
	if(p==nullptr){
		return;
	}
	// 前序遍历位置
	dfs(p->left);
	// 中序遍历位置
	dfs(p->right);
	// 后序遍历位置
}

// 非递归版前序遍历
void dfsPreO(TreeNode* head){
	if(head!=nullptr){
		stack<TreeNode*> st;
		st.push(head);
		while(!st.empty()){
			TreeNode *p=st.top();
			st.pop();
			cout<<p->val<<" ";
			// 后进先出，因此先让右端进栈
			if(p->right!=nullptr){
				st.push(p->right);
			}
			if(p->left!=nullptr){
				st.push(p->left);
			}
		}
	}
}

// 非递归版中序遍历
// 左中右 可以理解为每次遍历一棵子树的根节点
// 因此第一次经过节点会让该节点的左子树入栈
// 且站内只存放根节点与左子节点
void dfsInO(TreeNode* head){
	if(head!=nullptr){
		stack<TreeNode*> st;
		// head有可能是右节点，所以循环结束条件有两个
		while(!st.empty()||head!=nullptr){
			if(head!=nullptr){
				st.push(head);
				head=head->left;
			}
			else{
				head=st.top();
				st.pop();
				cout<<head->val<<" ";
				head=head->right;
			}
		}
	}
}

// 后序遍历非递归 单栈版
// 根据前序遍历更改入栈规则  中左右->中右左->左右中（翻转） 双栈实现
// 通过记录上一个遍历的节点可以优化掉储存栈
void dfsPostO(TreeNode* p){
	if(p!=nullptr){
		stack<TreeNode*> st;
		st.push(p);
		while(!st.empty()){
			TreeNode* cur=st.top();
			if(cur->left!=nullptr&&p!=cur->left&&p!=cur->right){
				st.push(cur->left);
			}
			else if(cur->right!=nullptr&&p!=cur->right){
				st.push(cur->right);
			}
			else{
				cout<<cur<<" ";
				p=cur;
				st.pop();
			}
		}
	}
}
```
- 先序中序和后序的输出的“并集”实际上就是递归序，反映了整个递归的过程
- 非递归可以通过自己定义一个栈实现递归遍历的操作
- 不论递归还是非递归，每个节点对栈操作三次，时间复杂度o(3n)，空间复杂度o(logn)(栈的深度)

```cpp
int minDepth(TreeNode* root) {
	if(root==nullptr){
		return 0;
	}
 	if(root->left==nullptr&&root->right==nullptr){
 		return 1;
	}
	int lh=MAXN,rh=MAXN;  // 防止空指针的污染

int minDepth(TreeNode* root) {
	if(root==nullptr) return 0;
	if(root->right==nullptr) return minDepth(root->left)+1;
	if(root->left==nullptr) return minDepth(root->right)+1;
	return min(minDepth(root->left),minDepth(root->right))+1;      
}  
```
### 判断是否为二叉搜索树
一棵二叉搜索树可被递归地定义为具有下列性质的二叉树：对于任一结点，
- 其左子树中所有结点的键值小于该结点的键值；
- 其右子树中所有结点的键值大于等于该结点的键值；
- 其左右子树都是二叉搜索树。
所谓二叉搜索树的“镜像”，即将所有结点的左右子树对换位置后所得到的树。
给定一个整数键值序列，现请你编写程序，判断这是否是对一棵二叉搜索树或其镜像进行前序遍历的结果。
#### 输入格式：
输入的第一行给出正整数 N（≤1000）。随后一行给出 N 个整数键值，其间以空格分隔。
#### 输出格式：
如果输入序列是对一棵二叉搜索树或其镜像进行前序遍历的结果，则首先在一行中输出 `YES` ，然后在下一行输出该树后序遍历的结果。数字间有 1 个空格，一行的首尾不得有多余空格。若答案是否，则输出 `NO`。
#### 输入样例 1：
```in
7
8 6 5 7 10 8 11
```
#### 输出样例 1：
```out
YES
5 7 6 8 11 10 8
```
#### 输入样例 2：
```in
7
8 10 11 8 6 7 5
```
#### 输出样例 2：
```out
YES
11 8 10 7 5 6 8
```
#### 输入样例 3：
```in
7
8 6 8 5 10 9 11
```
#### 输出样例 3：
```out
NO
```
#### 解法
```cpp
int n;
bool mirror=false;
vector<int> pre,post;

void getpost(int root,int tail){
	if(root>tail) return;
	int i=root+1,j=tail;
	if(!mirror){
		while(i<=tail&&pre[i]<pre[root]) i++;
		while(j>root&&pre[j]>=pre[root]) j--;
	}
	else{
		while(i<=tail&&pre[i]>=pre[root]) i++;
		while(j>root&&pre[j]<pre[root]) j--;
	}
	if(i-j!=1) return;
	getpost(root+1,j);
	getpost(i,tail);
	post.push_back(pre[root]);
}

void solve() {
	cin>>n;
	pre.resize(n);
	for(int i=0;i<n;i++){
		cin>>pre[i];
	}
	getpost(0,n-1);
	if(post.size()!=n){
		mirror=true;
		post.clear();
		getpost(0,n-1);
	}
	if(post.size()==n){
		cout<<"YES"<<'\n';
		for(int i=0;i<n;i++){
			cout<<(i==0?"":" ")<<post[i];
		}
	}
	else{
		cout<<"NO";
	}
}
```

### 判断是否为完全二叉树
https://leetcode.cn/problems/check-completeness-of-a-binary-tree/
1. 如果存在只有右孩子没有左孩子的树
2. 遇到左右孩子不全的节点后的所有节点都必须是叶子节点
```cpp
bool isCompleteTree(TreeNode* root) {
	if(root==nullptr) return true;
	queue<TreeNode*> q;
	q.push(root);
	bool ok=false;
	while(!q.empty()){
		TreeNode* cur=q.front();
		q.pop();
        if(cur->left!=nullptr){
            q.push(cur->left);
        }
        if(cur->right!=nullptr){
            q.push(cur->right);
        }
		if(ok){
			if(cur->left!=nullptr||cur->right!=nullptr){
				return false;
			}
		}
		else{
			if(cur->right!=nullptr&&cur->left==nullptr){
				return false;
			}
			if(cur->right==nullptr||cur->left==nullptr){
				ok=true;
			}
		}
	}
	return true;
}
```

## LCA最近共同祖先(LowestCommonAncestor)

- 已知p与q是同一个二叉树的不同节点，寻找二者最近的共同节点
- p和q存在两种关系：
	1. 一个恰好是另一个的祖先（包含关系）
	2. 二者分别属于同一棵树的不同子树
LCA有多种求解方式：
### 朴素DFS搜索
```cpp
TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
	if(root==nullptr||root==p||root==q){
		return root;
	}        
	TreeNode* l=lowestCommonAncestor(root->left,p,q);
	TreeNode* r=lowestCommonAncestor(root->right,p,q);
	if(l!=nullptr&&r!=nullptr){
		return root;
	}
	if(l==nullptr&&r==nullptr){
		return nullptr;
	}
	return l==nullptr?r:l;
}
```
### tarjan+并查集 离线批量解决LCA

```cpp
const int N = 5e5+5;

vector<int> adj[N];
vector<int> dsu;
vector<bool> vis;
vector<pair<int,int>> que[N];
int ans[N];
int n,m,s;

void build(){
	for(int i=0;i<=n;i++){
		dsu[i]=i;
	}
}

int find(int idx){
	if(dsu[idx]==idx) return idx;
	return dsu[idx]=find(dsu[idx]);
}

void merge(int x,int y){
	x=find(x);
	y=find(y);
	dsu[x]=y;
}

void tarjan(int u,int p){
	for(int v:adj[u]){
		if(v==p) continue;
		tarjan(v,u);
		merge(v,u);
	}
	vis[u]=true;
	for(auto [v,id]:que[u]){
		if(vis[v]){
			ans[id]=find(v);
		}
	}
}

void solve(){
	cin>>n>>m>>s;
	dsu.resize(n+1);
	vis.assign(n+1,0);
	build();
	for(int i=1;i<n;i++){
		int u,v;
		cin>>u>>v;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	for(int i=0;i<m;i++){
		int u,v;cin>>u>>v;
		que[u].push_back({v,i});
		que[v].push_back({u,i});
	}
	tarjan(s,0);
	for(int i=0;i<m;i++){
		cout<<ans[i]<<"\n";
	}
}

```
