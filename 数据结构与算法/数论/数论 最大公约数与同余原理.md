 // 最大公约数
 ll gcd(ll a,ll b){
	 return b==0?a:gcd(b,a%b);
 }

 ll lcm(ll a,ll b){
	 return a/gcd(a,b)*b;

## [878. 第 N 个神奇数字 - 力扣（LeetCode）](https://leetcode.cn/problems/nth-magical-number/)
```cpp
class Solution {
public:
    int nthMagicalNumber(ll n, ll a, ll b) {
        ll l=0,r=n*min(a,b),mid,ans=0;
	auto gcd=[](auto&& self,ll a,ll b)->ll{
		return (b==0?a:self(self,b,a%b));
	};
	auto check=[&](ll mid)->bool{
		return mid/a+mid/b-mid/(a*b/gcd(gcd,a,b))>=n;
	};
	while(l<=r){
		mid=(r-l)/2+l;
		// cout<<l<<" "<<r<<" "<<mid<<"\n";
		if(check(mid)){
			ans=mid%mod;
			r=mid-1;
		}
		else{
			l=mid+1;
		}
	}
    return ans;
    }
};
```

const ll MAXN = 2e18;
int n;
map<ll,ll> mp;
void solve(){
	cin>>n;
	ll ma=0;
	for(int i=0;i<n;i++){
		ll m;cin>>m;
		mp[m]++;
		ma=max(ma,m);
	}
	ll cidx=0,k=MAXN,idx=0;
	for(auto v:mp){
		if(v.first==idx){
			if(mp[idx]<k){
				k=mp[idx];
				cidx=idx;
			}
			idx++;
		}
		else{
			break;
		}
	}
	if(ma>idx){
		cout<<idx+1;
	}
	else{
		cout<<idx*k+cidx+1;
	}
}
