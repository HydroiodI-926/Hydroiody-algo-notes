# 埃氏筛
```cpp
vector<int> pr;
bool ip[N]; // N 查找质数的范围

void f(){
	fill(ip+2,ip+N,true);
	for(int i=2;i*i<=N;i++){
		if(ip[i]){
			for(int j=i*i;j<=N;j+=i){
				ip[j]=false;
			}
		}
	}
	for(int i=2;i<=N;i++){
		if(ip[i]){
			pr.push_back(i);
		}
	}
}
```

# 欧拉筛
[P3383 【模板】线性筛素数 - 洛谷](https://www.luogu.com.cn/problem/P3383)
```cpp
vector<int> pr;
bool vis[N];
 
void eulers(){
	vis[0]=vis[1]=true;
	for(int i=2;i<N;i++){
		if(!vis[i]){
			pr.push_back(i);
		}
		for(int j=0;j<pr.size();j++){
			if(i*pr[j]>N){
				break;
			}
			vis[i*pr[j]]=true;
			if(i%pr[j]==0){
				break;
			}
		}
	}
}
```
