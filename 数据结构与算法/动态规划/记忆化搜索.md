#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

const int N = 25;

int m,c;
vector<int> a[N];
vector<vector<int>> memo;

int dp(int idx,int mon){
    if(idx==c) return m-mon;
    if(memo[idx][mon]!=-1) return memo[idx][mon];
    int ans=memo[idx][mon];
    for(int v:a[idx]){
        if(mon>v){
            ans=max(ans,dp(idx+1,mon-v));
        }
    }
    return ans;
}

void solve() {
	cin>>m>>c;
	for(int i=0;i<c;i++){
        int n;cin>>n;
        a[i].resize(n);
        for(int j=0;j<n;j++){
            cin>>a[i][j];
        }
    }
    memo.assign(c,vector<int>(m+1,-1));
    int ans=dp(0,m);
    if(ans==-1){
        cout<<"no solution\n";
    }
    else{
        cout<<ans<<"\n";
    }
}

int main(){
	ios::sync_with_stdio(0);
	cin.tie(0),cout.tie(0);
	int T=1;
	cin>>T;
    while(T--){
    	solve();
    }
	return 0;
}
