1. 用空间代替重复计算，包含一整套原理和技巧的总和
2. 有些递归在展开计算时，总是重复调用同一个子问题的解，这种重复调用的递归变成动态规划收益高
3. 展开是不同解/重复调用的现象少，没有改成动态规划的必要
4. 因此任何动态规划的问题都一定对应这一个重复调用的行为的递归
5. 所以任何动态规划的问题都一定可以从递归入手，逐渐实现动态规划的方法（如斐波那契数列）
- 递推的实现思路：
	1. 能否容易得到初始状态（边界状态）的解
	2. 可以假设不大于n-1的状态已经解决
	3. 重点分析，规模扩大到n，如何枚举所有情况，然后用子问题状态f(1),f(2)...f(n-1)表示最终状态f(n)（状态转移）
- 做题思路：
1. 暴力递归尝试，自顶向底。但是纯暴力的递归或者dfs只能解决小数据量问题，会涉及到大量的重复计算
2. 基于递归改进，两种思考方向：
	- 记忆化搜索->从顶到底的动态规划，通过记忆化对递归过程进行剪枝，因此获得一定程度的优化
	- 递推求解   ->从底到顶的动态规划/严格位置依赖的动态规划
3. 决策\===状态转移方程==
	- **状态**：某种情况；状态的答案应该只依赖状态定义的情况和状态以外的常量。一般情况下，状态变化并不影响状态以外的量，当前状态之后的状态不受当前状态之前状态的影响。因此用动态规划解决问题的时候，应该把所有可能的变动加入状态的定义
	- **转移**：状态与状态之间的转换；
4. 严格位置依赖的动态规划 -> 优化枚举
	- 进一步优化空间（空间压缩），一维，二维，多维存在这种优化
	- 进一步优化时间
- dp的三个特点
	1. **子问题重叠性**：子问题会被反复计算多次，因此递归保存在数组中
	2. **无后效性**：当前子问题不受后面子问题影响，当前决策不影响后面的结构性质，即到达同一个状态时，无论走哪条路过来，未来的可能性完全相同
	3. **最优子结构**：下一阶段的最优解应该能由前面各阶段子问题的最优解导出，只保留与最终解集相关的部分代表信息
- 根据状态维度划分
	- 题设中有序列或者数组，那动态规划的状态就是一维的，通常用二维数组来储存状态
	- 题设中有棋盘或者网络，那么动态规划的状态就是二维的，通常用三维数组储存状态
	- 但是也不是唯一的，分析题目时，二维动态规划也有可能被压缩成一维动态规划

基于dfs记忆化搜索解
```cpp
const int MAXN = 1e9;
int n,m;

int dfs(int w,vector<int>& dp){
	if(w==0){
		return 0;
	}
	if(w<0){
		return MAXN;  // 剪枝，如果一种凑法超过了总数，那么这个答案是不合规的
	}
	if(dp[w]!=MAXN){
		return dp[w]; // 剪枝，记忆化搜索，由于是自顶向下的，越小的数额肯定会被面额大的纸币先被检索到，因此不用再更新
	}
	int x=min(dfs(w-1,dp),min(dfs(w-5,dp),dfs(w-11,dp)))+1;
	dp[w]=x;
	return dp[w];
}

void solve() {
    cin>>n>>m;
    vector<int> arr(n);
    vector<int> dp(m+1,MAXN);
    for(int i=0;i<n;i++) cin>>arr[i];
    cout<<dfs(m,dp);
}

```
但是这种方式即使通过数组实现对n个面额纸币dfs，很有可能会爆栈，尤其是m很大的时候，此时从自顶向底思考可以转换为自底向顶思考，即能不能从1开始填记忆数组，每次更新每个点的最小答案
```cpp
const int MAXN = 1e9;

int n,m;

void solve() {
    cin>>n>>m;
    vector<int> arr(n);
    vector<int> dp(m+1,MAXN);
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    dp[0]=0;
    for(int i=1;i<=m;i++){
    	for(int j=0;j<n;j++){
    		if(arr[j]<=i){ // 防止下面的比较时 re
    			// 已经记录方案与使用当前这个面额的纸币的方案比较
    			// 一开始数组标为MAXN，也就是最大值，表示没比较过
    			dp[i]=min(dp[i],dp[i-arr[j]]+1);
    		}
    	}
    }// o(m*n)
    cout<<dp[m];
}
```
## [P1216 数字三角形 Number Triangles - 洛谷](https://www.luogu.com.cn/problem/P1216)
##### 暴力解
通过观察题目发现有点类似二叉树的~~前序~~遍历，当然严格来讲这是个有向无环图，按照递归的思想，可以从最顶部开始递归，一直到最底部，是一种自顶向底的思路，因此用记忆化搜索可解
```cpp
int n;
int mp[1005][1005];
int dp[1005][1005];

int dfs(int x,int y){
	if(x>n||y>x){ // 越界，返回0
		return 0;
	}
	if(dp[x][y]!=-1){
		return dp[x][y]; // 没有越界，已经不是第一次来到这个点，直接返回结果
	}
	// 第一次来到这个点，更新这个点的信息，即这个点的值加上左右部分中最大的那个部分
	dp[x][y]=mp[x][y]+max(dfs(x+1,y),dfs(x+1,y+1));
	return dp[x][y];
}

void solve() {
    cin>>n;
    for(int i=1;i<=n;i++){
    	for(int j=1;j<=i;j++){
    		cin>>mp[i][j]; 
    		dp[i][j]=-1; // 标记可遍历的点
    	}
    }
    cout<<dfs(1,1);
}
```
##### 递推解
反过来想，是否可以从最底部开始一层层对比最大值结果，因此递推也可写
```cpp
int n;
int mp[1005][1005];
int dp[1005][1005];

void solve() {
    cin>>n;
    for(int i=1;i<=n;i++){
    	for(int j=1;j<=i;j++){
    		cin>>mp[i][j];
    	}
    }
    for(int j=1;j<=n;j++){
    	dp[n][j]=mp[n][j];
    }
    for(int i=n-1;i>=1;i--){
    	for(int j=1;j<=i;j++){
    		dp[i][j]=mp[i][j]+max(dp[i+1][j],dp[i+1][j+1]);
    	}
    }
    cout<<dp[1][1];
}
```

# 二维动态规划
1. dp表的规模 ：每个可变参数的可能性数量相乘
2. 时间复杂度估计：dp表大小 * 每个格子的枚举代价/决策代价
3. 空间压缩：二维dp表的空间往往很大，重新思考递推过程，看看有哪些空间是可以优化掉的，有些可以优化成滚动数组，甚至是一维dp表

## 例题
https://leetcode.cn/problems/minimum-path-sum/description/
```cpp
int minPathSum(vector<vector<int>>& grid){
	int n=grid[0].size();
	vector<int> dp(n,0);
	for(int i=0;i<m;i++){
		for(int j=0;j<n;j++){
            if(i=0) dp[j]=grid[i][j];
            else if(j=0) dp[j]=dp[j]+grid[i-1][j];
            else dp[j]=grid[i][j]+min(dp[j],dp[j-1]);
		}
	}
	return dp[n-1];
}
```
