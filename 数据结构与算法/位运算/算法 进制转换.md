### 十进制转二进制
#### 对于整数部分
每次除以2得到的余数就是二进制首位的数，1或0
#### 对于小数部分
每次乘以2得到的溢出的整数部分就是小数点后的数，1或0

```c++
#include <iostream>
#include <string>
using namespace std;

string decimalToBinary(double num) {
    string binary = "";
    
    // 处理整数部分
    int integerPart = (int)num;
    while (integerPart > 0) {
        binary = to_string(integerPart % 2) + binary;
        integerPart /= 2;
    }
    if (binary == "") binary = "0";
    
    // 处理小数部分
    binary += ".";
    double fractional = num - (int)num;
    
    for (int i = 0; i < 5; i++) {
        fractional *= 2;
        binary += to_string((int)fractional);
        fractional -= (int)fractional;
    }
    
    return binary;
}

int main() {
    double num;
    cout << "请输入十进制数: ";
    cin >> num;
    
    string result = decimalToBinary(num);
    cout << "二进制结果: " << result << endl;
    
    return 0;
}
```

| 名词              | 解释      |
| --------------- | ------- |
| decimalToBinary | 十进制转二进制 |
| binary          | 二进制     |
| integerPart     | 整数部分    |
| fractional      | 小数的     |
