- 基于贪心思想的单源最短路算法，用于解决带(正)权图的最短路问题,不能解决负权图问题
```cpp
// 模板
const ll INF = 1e15;
const int N = 1e5+5;

struct edge{
	int v;
	ll w;
};
struct node{
	ll d;
	int to;
	bool operator<(const node& other) const{
		return d>other.d;
	}
};
vector<edge> adj[N];
ll d[N];
bool vis[N];
int n,m,s;

void dijkstra(int s){
	fill(d+1,d+n+1,INF);
	d[s]=0;
	priority_queue<node> pq;
	pq.push({0,s});
	while(!pq.empty()){
		node cur=pq.top();
		pq.pop();
		int u=cur.to;
		if(vis[u]) continue;
		vis[u]=1;
		for(edge ed:adj[u]){
			int v=ed.v;
			ll w=ed.w;
			if(d[v]>d[u]+w){
				d[v]=d[u]+w;
				pq.push({d[v],v});
			}
		}
	}
}
```

const ll INF = 1e15;
const int N = 1e5+5;

struct node{
	ll d;
	int x,y;
	bool operator<(const node& other) const{
		return d>other.d;
	}
};
vector<vector<char>> mp;
vector<vector<ll>> d;
int n,m,sx,sy;
ll ans;
bool ok;

int dx[]={1,-1,0,0};
int dy[]={0,0,1,-1};

void dijkstra(int x,int y){
	d[x][y]=0;
	priority_queue<node> pq;
	pq.push({0,x,y});
	while(!pq.empty()){
		node cur=pq.top();
		pq.pop();
		int cx=cur.x,cy=cur.y;
		ll cw=cur.d;
		if(cw>d[cx][cy]) continue;
		if(mp[cx][cy]=='T'){
			ok=true;
			ans=cw;
			return;
		}
		for(int i=0;i<4;i++){
			int nx=cx+dx[i];
			int ny=cy+dy[i];
			if(nx<n&&nx>=0&&ny<m&&ny>=0&&mp[nx][ny]!='#'){
				ll nw=1;
				if(isdigit(mp[nx][ny])){
					nw+=(mp[nx][ny]-'0');
				}
				if(d[nx][ny]>d[cx][cy]+nw){
				d[nx][ny]=d[cx][cy]+nw;
				pq.push({d[nx][ny],nx,ny});
				}	
			}
		}
	}
}

void solve() {
	while(cin>>n>>m){
		ok=false;
	    mp.assign(n,vector<char>(m,'.'));
	    d.assign(n,vector<ll>(m,INF));
	    for(int i=0;i<n;i++){
	    	for(int j=0;j<m;j++){
	    		char c;
	    		cin>>c;
	    		if(c=='S'){
	    			sx=i;
	    			sy=j;
	    		}
	    		mp[i][j]=c;
	    	}
	    }
	    dijkstra(sx,sy);
	    if(ok) cout<<ans<<"\n";
	    else cout<<"impossible\n";
	}
}

# Bellman-Ford+SPFA优化
- 时间复杂度O($n*m$),因此只能在小图上使用
- 可以用来判负环，正常情况下最多只需要更新n-1次，如果超过了这个次数，说明有负环会无限使答案减小，spfa优化使用队列进行辅助，优化了常数时间
- **判负环关键**：统计每个节点被松弛的次数 `upd[v]`，当 `upd[v] >= n` 时存在负环
- **注意**：松弛条件 `d[u]+w < d[v]` 应独立于入队条件 `!vis[v]`，否则会漏更新已有节点的距离导致漏判负环
### [P3385 【模板】负环 - 洛谷](https://www.luogu.com.cn/problem/P3385)
```cpp
const ll MAXN = 2e18;
const int N = 2e3;

int n,m;

vector<pair<int,ll>> adj[N];
vector<ll> d;
vector<bool> vis;
vector<int> upd;

bool spfa(int s){
	queue<int> q;
	vis[s]=true;
	d[s]=0;
	q.push(s);
	while(!q.empty()){
		int u=q.front();
		q.pop();
		vis[u]=false;
		for(auto [v,w]:adj[u]){
			if(d[u]+w<d[v]){
				d[v]=d[u]+w;
				if(++upd[v]>=n) return true; // 检查 v 的松弛次数
				if(!vis[v]){
					vis[v]=true;
					q.push(v);
				}
			}
		}
	}
	return false;
}

void solve(){
	cin>>n>>m;
	d.assign(n+1,MAXN);
	upd.assign(n+1,0);
	vis.assign(n+1,false);
	for(int i=0;i<=n;i++) adj[i].clear();
	for(int i=0;i<m;i++){
		int u,v;
		ll w;
		cin>>u>>v>>w;
		adj[u].push_back({v,w});
		if(w>=0) adj[v].push_back({u,w});
	}
	cout<<(spfa(1)?"YES":"NO")<<"\n";
}
```

# Floyd算法-全源最短路
- 时间复杂度为O($n^3$),通过最朴素的动态规划，将所有边遍历一遍更新最小值，有n个点所以遍历n回
- 虽然整体慢，但是可以获取所有点的最短路，并且每个操作的常数时间小
- 使用floyd的时候点往往很少1e3左右
### [P2910 [USACO08OPEN] Clear And Present Danger S - 洛谷](https://www.luogu.com.cn/problem/P2910)

```cpp
const int INF = 1e9; // 无限大的数标记两个点不相邻/还未连通两点的距离

vector<vector<int>> adj; // 用邻接矩阵储存点与点的距离
vector<int> res;

int n,m;

void floyd(){
	// 最外层，选择桥，两点之间通过一个中间点作为桥（寻找有没有更短的路）
	for(int k=0;k<n;k++){ 
		// 起点
		for(int i=0;i<n;i++){
			// 终点
			for(int j=0;j<n;j++){
				// 起点与桥相通，桥与终点相通，新路距离比老路更短
				if(adj[i][k]!=INF&&adj[k][j]!=INF&&adj[i][j]>adj[i][k]+adj[k][j]){
				   	adj[i][j]=adj[i][k]+adj[k][j];
				}
			}
		}
	}
}

void solve(){
    cin>>n>>m;
    res.resize(m);
    adj.assign(n,vector<int>(n,INF));
    for(int i=0;i<m;i++){
    	cin>>res[i];
    }
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++){
    		cin>>adj[i][j]; // 一开始只记录相邻点的距离
    	}
    }
    floyd();
    ll ans=0;
    for(int i=0,j=1;j<m;i++,j++){
    	ans+=adj[res[i]-1][res[j]-1];
    }
    cout<<ans;
}
```
