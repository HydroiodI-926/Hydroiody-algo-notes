- 在**无向带权图**中选择一些边，在**保证连通性**的情况下，**边的总权和最小**
- 最小生成树可能不止一棵，只要保证是边的总权值最小的无环连通图，都是正确的
- 若无向带权图有n个点，那么最小生成树一定有n-1条边

# Kruskal算法 加边法 *常用*
1. 把所有边根据权值从小到大排序，从权值小的边开始考虑
2. 如果连接当前的边不会形成环，就选择当前边；反之不选
3. 考察完所有边之后（或是选择了n-1条边之后）可得最小生成树
- 以找边为主，更适合稀疏图
- 只需要建并查集，不需要建图
```cpp
int n,m;
struct edge{
	int u,v;
	ll w;
};
vector<int> father;
vector<edge> edges;

int find(int i){
	if(i!=father[i]){
		father[i]=find(father[i]);
	}
	return father[i];
}
// 并查集的合并操作，同时也检验了两个节点是否处于不同集合
// 处于不同集合，意味着此前没有连接（是否在同一个连通分量里面），则连接这条边
// 处于相同集合，意味着此前存在连接，再连接就会成环
bool unite(int x,int y){
	int rx=find(x);
	int ry=find(y);
	if(rx!=ry){
		father[rx]=ry;
		return true;
	}
	else{
		return false;
	}
}

bool comp(edge x,edge y){
	return x.w<y.w;
}

void solve(){
	cin>>n>>m;
	father.resize(n+1);
	edges.resize(m);
	// 并查集初始化，假设最初所有节点都没有相连
	for(int i=1;i<=n;i++){
		father[i]=i;
	}
	for(int i=0;i<m;i++){
		cin>>edges[i].u>>edges[i].v>>edges[i].w;
	}
	// 贪心，按权值从小到大对边排序，这样可以起到先把权值小的边利用起来的效果
	sort(edges.begin(),edges.end(),comp);
	ll ans=0;
	int cnt=0;
	for(edge x:edges){
		if(unite(x.u,x.v)){
			ans+=x.w;
			cnt++;
		}
	}
	if(cnt==n-1){
		cout<<ans;
	}
	else{
		cout<<"impossible";
	}
}
```

# Prim算法 加点法
1. 解锁的点的集合 set 解锁的边的集合 heap（小根堆）  set heap为空
2. 可以从任意点开始，起始点加入set，起始点所有边加入heap
3. 从heap中弹出权值最小的边e，查询e去往的点x
	- x在set中，e舍弃，重复步骤3
	- x不在set中，e启用，加入set，重复步骤3
4. heap为空时/set的数量与点的总数相同时，得到最小生成树
- 以找点为主，与边的多少无关，因此比Kruskal更适合稠密图
- 在添加节点的时候可以从任何点开始添加
- 需要建图
```cpp
#include <bits/stdc++.h>
typedef long long ll;
typedef unsigned long long ull;
using namespace std;
const ll INF = 2e9;
const int MAXN = 1005;

int n,m;
struct node{
	int next;
	ll w;
	bool operator<(const node& other) const{
		return w>other.w;
	}
};
vector<vector<node>> adj;
vector<bool> vis;

void solve(){
	cin>>n>>m;
	adj.assign(n+1,{});
	vis.assign(n+1,0);
	for(int i=0;i<m;i++){
		int u,v;
		ll w;
		cin>>u>>v>>w;
		adj[u].push_back({v,w});
		adj[v].push_back({u,w});
	}
	priority_queue<node> pq;
	for(node v:adj[1]){
		pq.push(v);
	}
	vis[1]=1;
	int ans=0,cnt=0;
	while(!pq.empty()){
		node edge=pq.top();
		pq.pop();
		int next=edge.next;
		ll cost=edge.w;
		if(!vis[next]){
			cnt++;
			vis[next]=1;
			ans+=cost;
			for(node v:adj[next]){
				if(!vis[v.next])
					pq.push(v);
			}
		}
	}
	if(cnt==n-1){
		cout<<ans;
	}
	else{
		cout<<"orz";
	}
}
int main(){
	ios::sync_with_stdio(0);
	cin.tie(0),cout.tie(0);
	solve();
	// int T;cin>>T;
    // while(T--){
    	// solve();
    // }
	return 0;
}
```
