## 随机快速排序改进--荷兰国旗问题
- 经典随机快速排序每次只能确定一个位置的数，改进后能确定与选定数的值相同的所有数


```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;
 

int n;
ll a[100005];

void q_sort(int l,int r){
	if(l>=r) return;
	int i=l,j=r;
	int mid=l+(r-l)/2;
	int p=a[mid];
	while(i<=j){
		while(a[i]<p) i++;
		while(a[j]>p) j--;
		if(i<=j){
			swap(a[i],a[j]);
			i++;j--;
		}
	}
	if(l<j) q_sort(l,j);
	if(i<r) q_sort(i,r);
}

int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	q_sort(0,n-1);
	for(int i=0;i<n;i++){
		cout<<a[i];
        if(i!=n-1) cout<<" ";
	}
	return 0;
} 

```
