# 源代码 o(nlogn)
```cpp
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
const int MAXN=1e6;

int arr[MAXN];
/*本质上是一个空间换时间，需要有o(n)的空间寄存一次归并的结果，压空间必然会增加时间*/
int help[MAXN]; 

// 归并操作
void merge(int l,int m,int r){
    int i=l,a=l,b=m+1;
    /*运用双指针对两个分块里的数一一归并排序*/
    while(a<=m&&b<=r){
        help[i++]=arr[a]<=arr[b]?arr[a++]:arr[b++];
    }
    /* 实际情况两个里面只有一个会触发，两个子块内部是有序的，
       因此最后有一个块剩下了2个以上的元素也可以直接接到辅助数组上*/
    while(a<=m){
        help[i++]=arr[a++];
    }
    while(b<=r){
        help[i++]=arr[b++];
    }
    /*copy操作将辅助数组上已经排好的块copy到原数组*/
    for(i=l;i<=r;i++){
        arr[i]=help[i];
    }
    // 可以看出来排序好的块与排序好的整体只差一个o(n)
}

// 递归写法
void mergeSort1(int l,int r){
    if(l==r){
        return;
    }
    int m=l+(r-l)/2;
    mergeSort1(l,m);
    mergeSort1(m+1,r);
    merge(l,m,r);
}

// 非递归写法
void mergeSort2(int n){
    for(int l,m,r,step=1;step<n;step<<=1){
        l=0;
        /*遍历到n结束*/
        while(l<n){  
            m=l+step-1;
            if(m+1>=n){ // 边界检测，如果m越界，则不进行归并
                break;
            }
            /*有边界有可能取的是分块的右端点，也有可能取到数组的边界，分块是手动分的
	            step==1：l=0,m=0,r=1   len=2
	            step==2：l=0,m=1,r=3   len=4
	            step==3：l=0,m=3,r=7   len=8
		    与递归的情况是一样的*/ 
            r=min(l+(step<<1)-1,n-1); 
            merge(l,m,r);
            l=r+1;
        }
    }
}

int main(){
    ios::sync_with_stdio(0);
	cin.tie(0),cout.tie(0);
    int n;cin>>n;
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    mergeSort1(0,n-1);
    // mergeSort2(n);
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    return 0;
}
```
