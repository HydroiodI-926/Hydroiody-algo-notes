# 模板O($n*n!$)
```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

vector<ll> interesting_nums;

// 预处理出所有的排列
void precompute() {
    // 长度从 1 到 9
    for (int len = 1; len <= 9; ++len) {
        string s = "";
        // 构造初始排列，例如长度为3时是 "123"
        for (int i = 1; i <= len; ++i) {
            s += to_string(i);
        }
        
        // 利用 next_permutation 生成该长度下的所有全排列
        do {
            interesting_nums.push_back(stoll(s));
        } while (next_permutation(s.begin(), s.end()));
    }
    // 注意：按上述顺序生成自然就是从小到大严格递增的，无需额外 sort
}

void solve() {
    ll l, r;
    cin >> l >> r;
    
    // 使用 lower_bound 二分查找第一个 >= l 的有趣数字
    auto it = lower_bound(interesting_nums.begin(), interesting_nums.end(), l);
    
    // 检查找到的数字是否存在，且是否 <= r
    if (it != interesting_nums.end() && *it <= r) {
        cout << *it << "\n";
    } else {
        cout << "-1\n";
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    precompute(); // 离线预处理
    
    int T;
    if (cin >> T) {
        while (T--) {
            solve();
        }
    }
    return 0;
}
```
